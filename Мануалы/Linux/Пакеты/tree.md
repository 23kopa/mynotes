```sh
# Показать дерево файлов с глубиной 2 уровня
tree -L 2

# Показать только каталоги
tree -d

# Показать все файлы, включая скрытые
tree -a

# Показать полный путь к файлам
tree -f

# Исключить файлы и каталоги, соответствующие маске (например, "*.txt")
tree -I "*.txt"

# Показать только файлы с расширением .jpg
tree -P "*.jpg"

# Показать размеры файлов в удобочитаемом формате (KB, MB, GB)
tree -h

# Показать размер каждого файла в байтах
tree -s

# Включить цветовую раскраску
tree -C

# Отключить вывод статистики о файлах и каталогах
tree --noreport

# Записать вывод в файл output.txt
tree -o output.txt

# Установить заголовок для вывода в формате HTML
tree -T "My Directory Structure" -H "."
```
