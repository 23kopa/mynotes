```bash 
#!/bin/bash

# Установка пакета dialog
apt-get install -y dialog

# Подтверждение установки Apache2
dialog --title "Установка Apache2" --yesno "Вы хотите установить пакет Apache2?" 7 60

# Получение кода возврата диалогового окна
response=$?
case $response in
  1) # Пользователь выбрал "Да"
    apt-get install -y apache2
    ;;
  2) # Пользователь выбрал "Нет"
    ;;
  3) # Пользователь нажал ESC или другую клавишу
    ;;
esac

# Подтверждение установки архиватора
dialog --title "Установка архиватора" --yesno "Вы хотите установить архиватор?" 7 60

response=$?
case $response in
  1) # Пользователь выбрал "Да"
    apt-get install -y zip
    ;;
  2) # Пользователь выбрал "Нет"
    ;;
  3) # Пользователь нажал ESC или другую клавишу
    ;;
esac

# Подтверждение установки сайта
dialog --title "Установка сайта" --yesno "Вы хотите установить сайт?" 7 60

response=$?
case $response in
  1) # Пользователь выбрал "Да"
    # Разархивация сайта
    unzip /путь/к/архиву.zip -d /путь/к/разархивированному/сайту

    # Перенос сайта на сервер
    cp -r /путь/к/разархивированному/сайту /var/www/html/
    ;;
  2) # Пользователь выбрал "Нет"
    ;;
  3) # Пользователь нажал ESC или другую клавишу
    ;;
esac

# Уведомление об успешной установке
dialog --msgbox "Установка завершена успешно!" 7 60

# Завершение установки
clear




Вам необходимо сохранить этот скрипт в файл, например installer.sh, и сделать его исполняемым с помощью команды chmod +x installer.sh. Затем вы можете запустить его с правами суперпользователя (sudo) для установки пакетов и выполнения необходимых операций:

sudo ./installer.sh
```
